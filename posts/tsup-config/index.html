Configure tsup to build typescript library


// package.json

{
  "script": {
    "build": "tsup --out-dir dist"
  },
  "tsup": {
    "clean": true,
    "sourcemap": false,
    "dts": true,
    "minify": true,
    "target": "ES2020",
    "format": [
      "esm",
      "cjs"
    ],
    "entry": [
      "src/index.ts"
    ]
  },
}
